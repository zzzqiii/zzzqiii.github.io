使用forcats处理因子

library(tidyverse)
library(forcats)

创建因子

要想创建一个因子，必须先创建有效水平的一个列表：

x1 <- c("Dec", "Apr", "Jan", "Mar")
month_levels <- c(
"Jan", "Feb", "Mar", "Apr", "May", "Jun",
"Jul", "Aug", "Sep", "Oct", "Nov", "Dec"
)
y1 <- factor(x1, levels = month_levels)
y1

## [1] Dec Apr Jan Mar
## 12 Levels: Jan Feb Mar Apr May Jun Jul Aug ... Dec

sort(y1)

## [1] Jan Mar Apr Dec
## 12 Levels: Jan Feb Mar Apr May Jun Jul Aug ... Dec

x2 <- c("Dec", "Apr", "Jam", "Mar")
y2 <- factor(x2, levels = month_levels)
y2

## [1] Dec  Apr  <NA> Mar 
## 12 Levels: Jan Feb Mar Apr May Jun Jul Aug ... Dec

#不在有效水平集合内的所有值都会自动转换为NA

#显示错误信息
y2 <- parse_factor(x2, levels = month_levels)

## Warning: 1 parsing failure.
## row col           expected actual
##   3  -- value in level set    Jam

#如果省略了定义水平的这个步骤，那么会将按字母顺序排序的数据作为水平
factor(x1)

## [1] Dec Apr Jan Mar
## Levels: Apr Dec Jan Mar

#如果要让因子的顺序与初始数据的顺序保持一致，在创建因子时，将水平设置为unique(x)，或者在创建因子后再对其使用fct_inorder() 函数。
f1 <- factor(x1, levels = unique(x1))
f1

## [1] Dec Apr Jan Mar
## Levels: Dec Apr Jan Mar

f2 <- x1 %>% factor() %>% fct_inorder()
f2

## [1] Dec Apr Jan Mar
## Levels: Dec Apr Jan Mar

#如果想要直接访问因子的有效水平集合，那么可以使用levels() 函数
levels(f2)

## [1] "Dec" "Apr" "Jan" "Mar"

修改因子水平

fct_recode() 函数可以对每个水平进行修改或重新编码

gss_cat %>% count(partyid)

## # A tibble: 10 x 2
##    partyid                n
##  * <fct>              <int>
##  1 No answer            154
##  2 Don't know             1
##  3 Other party          393
##  4 Strong republican   2314
##  5 Not str republican  3032
##  6 Ind,near rep        1791
##  7 Independent         4119
##  8 Ind,near dem        2499
##  9 Not str democrat    3690
## 10 Strong democrat     3490

gss_cat %>% 
  mutate(partyid = fct_recode(partyid, 
                              "Republican, strong" = "Strong republican",
"Republican, weak" = "Not str republican",
"Independent, near rep" = "Ind,near rep",
"Independent, near dem" = "Ind,near dem",
"Democrat, weak" = "Not str democrat",
"Democrat, strong" = "Strong democrat")) %>%
  count(partyid)

## # A tibble: 10 x 2
##    partyid                   n
##  * <fct>                 <int>
##  1 No answer               154
##  2 Don't know                1
##  3 Other party             393

#fct_recode() 会让没有明确提及的水平保持原样，也可以将多个原水平赋给同一个新水平，这样就可以合并原来的分类
gss_cat %>%
  mutate(partyid = fct_recode(
    partyid,
    "Republican, strong" = "Strong republican",
"Republican, weak" = "Not str republican",
"Independent, near rep" = "Ind,near rep",
"Independent, near dem" = "Ind,near dem",
"Democrat, weak" = "Not str democrat",
"Democrat, strong" = "Strong democrat",
"Other" = "No answer",
"Other" = "Don't know",
"Other" = "Other party"
  )) %>%
  count(partyid)

## # A tibble: 8 x 2
##   partyid                   n
## * <fct>                 <int>

## 2 Republican, strong     2314

#如果想要合并多个水平，那么可以使用fct_recode() 函数的变体fct_collapse() 函数。
gss_cat %>%
  mutate(partyid = fct_collapse(
    partyid,
    other = c("No answer", "Don't know", "Other party"),
rep = c("Strong republican", "Not str republican"),
ind = c("Ind,near rep", "Independent", "Ind,near dem"),
dem = c("Not str democrat", "Strong democrat")
  )) %>%
  count(partyid)

练习

美国民主党、共和党和中间派的人数比例是如何随时间而变化的？

levels(gss_cat$partyid)

##  [1] "No answer"          "Don't know"        
##  [3] "Other party"        "Strong republican" 
##  [5] "Not str republican" "Ind,near rep"      
##  [7] "Independent"        "Ind,near dem"      
##  [9] "Not str democrat"   "Strong democrat"

gss_cat %>%
  mutate(partyid = fct_collapse(
    partyid,
    other = c("No answer", "Don't know", "Other party"),
rep = c("Strong republican", "Not str republican"),
ind = c("Ind,near rep", "Independent", "Ind,near dem"),
dem = c("Not str democrat", "Strong democrat")
  )) %>%
  count(year, partyid) %>%
  group_by(year) %>%
  mutate(p = n / sum(n)) %>%
  ggplot(aes(
    x = year, y = p,
    color = fct_reorder2(partyid, year, p)
  )) +
  geom_point() +
  geom_line() +
  labs(color = "Party ID")
