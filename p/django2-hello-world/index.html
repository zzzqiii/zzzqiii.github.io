编辑urls.py文件，规定有效访问路径

from django.contrib import admin
from django.urls import path
from . import views

urlpatterns = [
    path('admin/', admin.site.urls),
    path('', views.index),
]


from django.contrib import admin
from django.urls import re_path
from . import views

urlpatterns = [
    path('admin/', admin.site.urls),	#规定后台管理的方式http://127.0.0.1:8000/admin/
    path('^$', views.index),  #127.0.0.1:8000
]
