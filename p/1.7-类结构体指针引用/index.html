        }

        {
            return age;
        }

        {
            money += x;
        }
       
       
    //可以在外面也可以在类里面调用
   
    //可以有多个private和public
}b, persons[100];//也可以直接在后面声明。 
//一定要加分号
struct Person
{
    int age, height;
    double money;
    Person() {}
   
    Person(int _age, int _height, double _money) //构造函数，和结构体同名，没有类型，里面可以传递参数。可以定义多个构造函数。
    {
        age = _age;
        height = _height;
        money = _money;
    }
   
   
    //特殊写法，更快。
    Person(int _age, int _height, double _money) : age(_age), height(_height), money(_money) {}
   
};
int main()
{
    Student c;
    Person people[50]; //也可以声明数组
    Person p(18, 180, 100.0); //调用结构体的构造函数。当直接定义变量没有给初值的时候，要写一个空的构造函数。
    Person x = {18, 189, 100}; //用数组的方式初始化。按定义的顺序来复制。
    Person y = {18, 179}; //如果没有被定义完，那么没有定义的的默认为0；
    c.name = "zzq";
    //c.age = 18; 错误
   
    cout << c.get_age() << endl;
    c.add_money(1000);
    return 0;
}

指针，引用

栈的空间从上往下开，堆从下往上。

局部变量定义在栈里面，静态变量和数组定义在堆中。

局部变量不确定的，开到堆中的会初始化为0.

原值，地址：a数组存到内存的下标中。

找地址

char c = 'a';
cout << (void*)&c << endl;

int a = 10;

int* p = &a; p存的值就是a的地址。这里的*表示变量类型是指针。

cout « *p « endl; 先看变量p，想求变量p的值的时候，在前面加一个*, *使一个操作符，和！一样（感叹号表示取反）。

*p = 12; 赋值

cout « a « endl; 输出12，地址上的值发生了变化，因此a也会发生相应的变化。

指针相当于数组下标。

int** = &p 求指针的指针。

数组也是指针。

int a[5] = {1, 2};

cout « a « endls;

会输出a的指针。

for (int i = 0; i < 5; i++)

​ cout « (void*)&a[i] « endl;

a和a[0]的地址是一样的。

数组名字就是一个特殊的指针，这个指针的值就是数组中第一个变量的地址。

数组在地址里是连续的一段。

如果是chr类型的，直接输出数组名字会显示字符串，要进行强制转换。

数组除了存了第一个变量的地址，还存了数组的长度。

指针支持运算。

int a[5] = {1, 2, 3, 4, 5};

int *p = a;

cout « p « endl;

cout « p + 1 « endl; 输出下一个地址。比上一个加4，因为数组是int型的，如果是chr型的话加1.

如果是long long的话差8. +1不是数值加1，而是跳到数组的下一个变量，跳的长度是变量的长度。。

访问a[2]还可以用p+2

cout « * (p+2 ) « endl; 输出a[2]

因为数组本身就是一个指针，所以可以直接用a来写：

cout « *(a + 2) « endl;

int *x = &a[0], *y = &a[2];

cout « x - y « endl; 输出2。 做的不是数值的减法，而是两个数值的差除以变量的长度。

用一个变量来操作另一个变量：

int *p = &a;

简化为int &p = a; 引用、别名。定义了一个变量p，p和a存在同一个位置，用的同一个地址。

链表

链表是算法中的一种结构。

链表在定义的时候使一个结构体，里面存两个变量，变量（数据值）和指针：

struct Node
{
int val;
Node* next; //next指针指向另外一个node的地址。
p->next = p; //把下一个指针定义为p自己
};

如果p是指针，要用->来调用成员变量。p->next = p;

如果p是变量，用.来调用。p.next = p;

new: 动态开辟一个结构体

new：动态开辟运算空间
