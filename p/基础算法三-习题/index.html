双指针算法

最长不连续重复子序列

#include <iostream>

using namespace std;

const int N = 100010;

int a[N], s[N];

int main()
{
    int n;
    cin >> n;
    int res = 0;
    for (int i = 0; i < n; i ++)
        cin >> a[i];
    for (int i = 0, j = 0; i < n; i ++)
    {
        s[a[i]] ++;
        while (s[a[i]] > 1)
        {
            s[a[j]] --;
            j ++;
        }
        res = max(res, i - j + 1);
    }
    cout << res;
    return 0;
}

数组元素的目标和

#include <iostream>

using namespace std;

const int N = 100010;
int n, m, x;
int a[N], b[N];

int main()
{
    cin >> n >> m >> x;
    for (int i = 0; i < n; i ++)    cin >> a[i];
    for (int j = 0; j < m; j ++)    cin >> b[j];
    for (int i = 0, j = m - 1; i < n; i ++)
    {
        while (j >= 0 && a[i] + b[j] > x)   j --;
        if (a[i] + b[j] == x)  
        {
            printf("%d %d\n", i, j);
            break;
        }
    }
    return 0;
}

位运算

二进制中1的个数

#include <iostream>

using namespace std;

int lowbit(int x)
{
    return x & -x;
}

int main()
{
    int n;
    cin >> n;
    while (n --)
    {
        int m;
        cin >> m;
        int res = 0;
        while (m)
        {
            m -= lowbit(m);
            res += 1;
        }
        cout << res << ' ';
    }
    return 0;
}

64位整数乘法

求 a 乘 b 对 p 取模的值。

输入格式 第一行输入整数 a ，第二行输入整数 b ，第三行输入整数 p 。

输出格式 输出一个整数，表示a*b mod p的值。

数据范围 1≤a,b,p≤1018 输入样例： 3 4 5 输出样例： 2

#include <iostream>
#define ll long long

using namespace std;

int main()
{
    ll a, b, p, res = 0;
    cin >> a >> b >> p;
    while (b)
    {
        if (b & 1)  res = (res + a) % p; //判断最后一位是否为1，如果是1，那么在加法分配律里应该乘的是1不是0；
        //每一步循环的操作是加上再对p求模。
        b >>= 1;
        a = a * 2 % p;
    }
    cout << res << endl;
    return 0;
}

区间和（离散化）

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int N = 300010;

int a[N], s[N];

vector<int> alls;

typedef pair<int, int> PII;
vector<PII> add, query;

int find(int x)
{
    int l = 1, r = alls.size();
    while (l < r)
    {
        int mid = l + r >> 1;
        if (alls[mid] >= x) r = mid;
        else l = mid + 1;
    }
    return r;
    
    //return lower_bound(alls.begin, alls.end(), x) - alls.begin() + 1;
}

int main()
{
    int n, m;
    cin >> n >> m;
    
    for (int i = 0; i < n; i ++)
    {
        int x, c;
        cin >> x >> c;
        add.push_back({x, c});
        alls.push_back(x);
    }
    
    for (int i = 0; i < m; i ++)
    {
        int l, r;
        cin >> l >> r;
        query.push_back({l, r});
        alls.push_back(l);
        alls.push_back(r);
    }
    
    sort(alls.begin(), alls.end());
    alls.erase(unique(alls.begin(), alls.end()), alls.end());
    
    for (auto item : add)
    {
        int x = find(item.first);
        a[x] += item.second;
    }
    
    for (int i = 1; i <= alls.size(); i ++)   s[i] = s[i - 1] + a[i];
    
    for (auto item : query)
    {
        int l = find(item.first), r = find(item.second);
        cout << s[r] - s[l - 1] << endl;
    }
    
    return 0;
}

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef pair<int, int> PII; //操作，有两个数。

const int N = 300010;

int n, m;
int a[N], s[N]; //a数组是存插入后该坐标的值，s数组是前缀和。

vector<int> alls; //存入的所有要离散化的值。
vector<PII> add, query; //插入和查询都需要两个数，所以用pair

//找原本的坐标离散化后的值
int find(int x)
{
    int l = 0, r = alls.size() - 1;
    while (l < r)
    {
        int mid = l + r >> 1;
        if (alls[mid] >= x) r = mid;
        else l = mid + 1;
    }
    return r + 1; //因为要用到前缀和，所以映射到从1开始的自然数。
}

int main()
{
    int n, m;
    cin >> n >> m;
    //先把所有需要操作的数读进来。（n次操作）
    for (int i = 0; i < n; i ++)
    {
        int x, c;
        cin >> x >> c;
        add.push_back({x, c}); //对x位置加上c；
        
        alls.push_back(x);
    }
    for (int i = 0; i < m; i ++)
    {
        int l, r;
        cin >> l >> r;
        query.push_back({l, r}); //查询l和r
        alls.push_back(l);
        alls.push_back(r);
    }
    //去重
    sort(alls.begin(), alls.end());
    alls.erase(unique(alls.begin(), alls.end()), alls.end());
    
    //处理插入
    for (auto item : add)
    {
        int x = find(item.first); //求离散化后的值
        a[x] += item.second;
    }
    
    //预处理前缀和
    for (int i = 1; i <= alls.size(); i ++)  s[i] = s[i - 1] + a[i];
    
    //处理询问
    for (auto item : query)
    {
        int l = find(item.first), r = find(item.second);
        cout << s[r] - s[l - 1] << endl;
    }
}

vector<int>::iterator unique(vector<int> &a)
{
	int j = 0;
    for (int i = 0; i < a.size(); i ++)
        if (!i || a[i] != a[i - 1])
            a[j ++] = a[i];
    //a[0] ~ a[j - 1]所有a中不重复的数
    return a.begin() + j;
}

unique函数的实现

美团面试题，格子染色 acwing759

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef pair<int, int> PII;

vector<PII> segs;

void merge(vector<PII> &segs)
{
    int st = -2e9, ed = -2e9;
    vector<PII> res;
    sort(segs.begin(), segs.end()); //忘记排序好多次
    for (auto seg : segs)
    {
        if (ed < seg.first)
        {
            if (st != -2e9) res.push_back({st, ed}); //pair忘记加大括号好多次
            st = seg.first, ed = seg.second; //second, first后面不用加括号
        }
        else ed = max(ed, seg.second);
    }
    if (st != -2e9) res.push_back({st, ed});
    segs = res;
}

int main()
{
    int n;
    cin >> n;
    while (n --)
    {
        int l, r;
        cin >> l >> r;
        segs.push_back({l, r});
    }
    merge(segs);
    cout << segs.size() << endl;
    return 0;
}
