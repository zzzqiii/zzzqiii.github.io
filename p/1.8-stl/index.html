vector

#include <iostream>
#include <algorithm>
#include <vector>
#include <ctime>

using namespace std;

bool cmp(int a, int b) //a是否应该排在b的前面
{
    return a < b; //如果a小于b的话a就应该排在b的前面。
}

int main()
{
    vector<int> a({1, 2, 3, 4, 5});
    
    
    reverse(a.begin(), a.end());
    
    for (int x : a) cout << x << ' ';
    
    cout << endl;
    
    random_shuffle(a.begin(), a.end()); //把数组的顺序打乱。但是每次运行的结果是一样的。
    srand(time(0));//加了一个跟时间有关的随机数种子，再打乱之后每次运行的结果就会发生变化。
    random_shuffle(a.begin(), a.end());
    
    sort(a.begin(), a.end());//从小到大
    sort(a.begin(), a.end(), greater<int>()); //从大到小
    sort(a.begin(), a.end(), cmp); //自定义比较。
    
    for (int x : a) cout << x << ' ';
    cout << endl;
    
    
    
    int b[] = {1, 2, 4, 5, 6};
    reverse(b, b + 5); //逗号前写第一个位置的指针，逗号后写最后一个的下一个。
    for (int x : b) cout << x << ' ';
    
    cout << endl;
    
    int p[] = {1, 1, 2, 2, 3, 3, 4};
    int m = unique(p, p+7) - p;
    cout << m << endl;
    
    for (int i = 0; i < m; i ++) cout << p[i] << ' ';
    cout << endl;
    
    vector<int> q({1,1,2,2,3,3,4});
    int n = unique(q.begin(), q.end()) - q.begin();
    cout << n << endl;
    for (int i = 0; i < n; i ++) cout << q[i] << ' ';
    cout << endl;
    
    q.erase(unique(q.begin(),q.end()), q.end()); //把重复的元素去除后保留下来，不重复的删除。
    for (int i = 0; i < n; i ++) cout << q[i] << ' ';
    cout << endl;
    
    int z[] = {1, 2, 6, 4, 5};
    int *l = lower_bound(z, z + 5, 3);  //返回大于等于3的第一个值
    cout << *l << endl;
    int r = lower_bound(z, z + 5, 7) - z; //返回end() //返回大于等于3的第一个值
    cout << r << endl;
    
    
    
    return 0;
}

队列

#include <iostream>
#include <vector>
#include <queue>

using namespace std;

int main()
{
    queue<int> q;
    queue<double> a;
    struct Rec
    {
        int a, x, y;
    }
    queue<Rec> b;
    //定义队列。循环队列，先进先出。
    //优先队列，优先往外弹最大的数。
    
    priority_queue<int> a;//大根堆
    priority_queue<int, vector<int>, greater<int>> b; //小根堆
    priority_queue<pair<int, int>> c;
    
    struct Rec
    {
        int a, b;
        bool operator< (const Rec& t) const
        {
            return a < t.a;
        }
    } //优先队列要重载小于号.
    //如果用的是小根堆，那么要重载大于号。
    priority_queue<Rec> d;
    
    q.push(1);//在队头插入一个元素。只能在队头插入
    q.pop();//弹出队尾元素。只能在队尾弹出。
    q.front();//返回队头
    q.back();//返回队尾
    
    priority_queue<int> a;
    a.push(1); //插入一个数（位置不确定）
    a.top(); //取最大值
    a.pop();//删除最大值
    
    //队列没有clear函数。
    q = queue<int>(); //重新初始化一个队列，相当于清空
    
}

栈stack

#include <iostream>
#include <vector>
#include <queue>
#include <stack>
#include <deque>

using namespace std;

int main()
{
    stack<int> stk; //定义栈. 先进后出
    stk.push(1); //插入一个元素
    stk.top(); //返回栈顶元素
    stk.pop();//删除栈顶元素。
    //队列只能在队头弹出，队尾插入
    //栈只能从队尾插入，只能从队尾弹出。
    //双端队列没有限制，两端都是可进可出的。
    
    //deque在队头队尾插入都是O(1)的。
    
    deque<int> a;
    a.begin(), a,end();
    a.front(), a.back();
    
    a.push_back(1), a.push_front(2);
    a[0];
    a.pop_back(), a.pop_front();
    
    a.clear();
    return 0;
    
}

set

#include <iostream>
#include <vector>
#include <queue>
#include <stack>
#include <deque>
#include <set>
#include <unordered_set> //无序的set。效率更高，但是不支持二分。
#include <bitset>

using namespace std;

int main()
{
    set<int> a; //元素不能重复
    multiset<int> b; //元素可以重复。
    
    struct Rec
    {
        int x, y;
        bool operator< (const Rec& t) const
        {
            return x < t.x;
        }
    }
    set<Rec> c;
    
    set<int>::iterator it = a.begin();//定义迭代器
    it ++;
    it --;
    ++ it;
    -- it;
    
    a.end();
    
    a.insert(x);
    a.find(x); 
    if (a.find(x) == a.end()) //判断x在a中是否存在。没找到返回a.end(),找到了返回一个迭代器。
    
    a.lower_bound(x); //找到大于等于x的最小元素的迭代器。
    a.upper_bound(x); //找到大于x的最小元素的迭代器。
    
    a.erase(x); //把所有迭代器删掉
    a.erase(it); //把it这个迭代器删掉。
    
    a.count(x); //返回集合s中等于x的个数。因为它是不重复的，所以存在的话返回1，不存在返回0；multiset可以重复，所以可以大于1.
    
    //迭代器可以理解成指针，作用和循环的作用类似。
    
    unordered_set<int> S; //哈希表。和set完全一样的用法，只是没有lower_bound和upper_bound的操作。
    unordered_multiset<int> b; //哈希表，可以存储重复元素。
    
    
    bitset<1000> a; //定义一个长度为一千位的01串。
    a[0] = 1;//赋值。没有赋值的为0；
    a.count(); //返回1的个数。
    a.set(3); //把第3位设成1.
    a.reset(3); //把第3位设成0。
    
    
    pair<int, string> a, b; //一个二元组。
    a = {3, "zzq"};
    a = make_pair(4, "abc"); //赋值
    cout << a.first << ' ' << a.second << endl;//输出这俩个值。
    a > b;//pair支持比较运算。双关键字比较。
    //同理vector也支持比较。按照字典序比较。
}

map

#include <iostream>
#include <vector>
#include <queue>
#include <stack>
#include <deque>
#include <set>
#include <map>
#include <unordered_map> //效率更高，但不支持二分。尽可能用这个

using namespace std;

int main()
{
    map<int, int> a; //map存的是一个二元组，把第一个元素映射到第二个元素。
    a[1] = 2;
    a[100000] = 3;
    
    cout << a[100000] << endl;
    
    map<string, int> a;
    a["zzq"] = 2;
    cout << a["zzq"] << end; //相当于python中字典的键和值
    
    map<string, vector<int>> a;
    a["zz"] = vector<int>({1, 2, 3, 4});
    cout << a["zz"].size() << endl;
    cout << a["zz"][2] << endl;
    
    a.insert({"a", {1, 2}}); //插入
    a.find("zzq"); //传入一个key
    
    
    return 0;
}
